```typescript
const handleBulkTagConfirm = async () => {
  const targetId = savedAgentIdRef.current;
  if (!targetId) {
    router.push("/agentes");
    return;
  }

  setApplyingTags(true);
  try {
    const res = await fetch(`/api/agents/${targetId}/tags`, {
      method: "POST",
    });

    if (res.ok) {
      const data = await res.json();
      toast.success(`Tag aplicada em ${data.affected} chats`);
    } else {
      toast.error("Erro ao aplicar tags");
    }
  } catch {
    toast.error("Erro ao aplicar tags");
  } finally {
    setApplyingTags(false);
    setShowBulkTagDialog(false);
    router.push("/agentes");
  }
};
```

Key changes:
- Uses `savedAgentIdRef.current` instead of `agentId` prop — this correctly picks up the newly created agent's ID in create mode
- Removes the `!agentId && mode === "create"` early return that was silently skipping the bulk tag POST
- The guard `if (!targetId)` is a safety fallback only — it should never trigger because onSubmit already checks `savedAgentId` before showing the dialog
